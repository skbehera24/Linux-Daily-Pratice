grep (Global Regular Expression Print) command
--------------------------------------------------


sumit@ubuntu:~/behera$ cat njo.txt
He is a boy.
May be he gone.
I have a pen.
I will go to office.
Santos is good boy.
Sumit is going to school.
I am alone.

sumit@ubuntu:~/behera$ grep 'is' njo.txt
He is a boy.
Santos is good boy.
Sumit is going to school.




//to avoid case sensitive we use -i with grep command

sumit@ubuntu:~/behera$ grep  -i 'iS' njo.txt
He is a boy.
Santos is good boy.
Sumit is going to school.
sumit@ubuntu:~/behera$



// to know on which line it is we use -ni


sumit@ubuntu:~/behera$ grep  -i 'iS' njo.txt
He is a boy.
Santos is good boy.
Sumit is going to school.
sumit@ubuntu:~/behera$ grep  -ni 'iS' njo.txt
1:He is a boy.
5:Santos is good boy.
6:Sumit is going to school.



sumit@ubuntu:~/behera$ grep  -in 'iS' njo.txt
1:He is a boy.
5:Santos is good boy.
6:Sumit is going to school.
sumit@ubuntu:~/behera$


//to ignore data we use -v  
sumit@ubuntu:~/behera$ grep  -v 'is' njo.txt
May be he gone.
I have a pen.
I will go to office.
I am alone.

//to avoid case sensitive we use -i



sumit@ubuntu:~/behera$ grep  -v 'is' njo.txt
May be he gone.
I have a pen.
I will go to office.
I am alone.

sumit@ubuntu:~/behera$ grep  -vi 'is' njo.txt

ZIP files in Linux
-----------------------------
sumit@ubuntu:~/behera$ zip file bmk.txt hlo.txt njo.txt
  adding: bmk.txt (stored 0%)
  adding: hlo.txt (stored 0%)
  adding: njo.txt (deflated 24%)


sumit@ubuntu:~/behera$ ls

sumit@ubuntu:~/behera$ zip omghdk bmk.txt hlo.txt njo.txt
  adding: bmk.txt (stored 0%)
